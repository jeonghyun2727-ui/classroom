async function getBalance(studentNum){
  const s=allStu.find(x=>String(x.num)===String(studentNum));
  if(!s)return{ok:false};
  const txs=await sb(`ledger?student_num=eq.${studentNum}&select=*&order=id.asc`);
  let balance=s.init_bal||0;
  const result=txs.map(r=>{
    const isIn=['주급 지급','보너스 지급'].includes(r.tx_type);
    if(isIn)balance+=r.amount;else balance-=r.amount;
    return{date:r.created_at,type:r.tx_type,item:r.item,amount:r.amount,inOut:isIn?'+':'-',balance,memo:r.memo};
  });
  return{ok:true,balance,initBal:s.init_bal||0,job:s.jobs||'',weekly:s.weekly||0,name:s.name,txs:result};
}
async function getAllBalances(){
  const ledger=await sb('ledger?select=*');
  return allStu.map(s=>{
    const num=String(s.num);
    let balance=s.init_bal||0,earned=0,spent=0;
    ledger.filter(r=>String(r.student_num)===num).forEach(r=>{
      const amt=r.amount||0;
      if(['주급 지급','보너스 지급'].includes(r.tx_type)){balance+=amt;earned+=amt;}
      else{balance-=amt;spent+=amt;}
    });
    return{num:s.num,name:s.name,job:s.jobs||'',weekly:s.weekly||0,initBal:s.init_bal||0,balance,earned,spent};
  });
}

  document.getElementById('stu-wk').textContent=(res.weekly||0).toLocaleString();
  document.getElementById('stu-earn').textContent=earn.toLocaleString()+'별';
  document.getElementById('stu-spent').textContent=spent.toLocaleString()+'별';
  // 알림 배너
  const ledger=await sb(`ledger?student_num=eq.${CU.num}&select=tx_type,created_at`);

async function loadGroupTickets(){

  const gts=await sb('group_tickets?select=*&order=created_at.desc');

        +'<span style="font-size:.78rem;color:var(--purple);font-weight:700;background:#eeeeff;padding:2px 8px;border-radius:8px">'+gt.status+'</span>'
      +'</div>'
      +(gt.description?'<div style="font-size:.83rem;color:#888;margin-bottom:8px">'+gt.description+'</div>':'')
      +'<div style="background:#f0f0f0;border-radius:20px;height:10px;overflow:hidden;margin-bottom:6px">'
        +'<div style="background:'+(done?'var(--gold)':'var(--purple)')+';height:100%;width:'+pct+'%;border-radius:20px"></div>'
      +'</div>'
      +'<div style="display:flex;justify-content:space-between;font-size:.85rem">'
        +'<span style="color:var(--purple);font-weight:700">'+total.toLocaleString()+' 별 모였어요</span>'

async function loadGroupTicketsTeacher(){
  const el=document.getElementById('gtListTeacher');
  el.innerHTML='<div class="loading"><div class="spin"></div></div>';
  const gts=await sb('group_tickets?select=*&order=created_at.desc');
  const parts=await sb('group_ticket_participants?select=*');
  if(!gts.length){el.innerHTML='<div class="empty">공동구매가 없어요</div>';return;}
  let html='';
  for(const gt of gts){
    const myParts=parts.filter(p=>p.group_id===gt.id);
    const total=myParts.reduce((s,p)=>s+p.amount,0);
    const pct=gt.goal>0?Math.min(100,Math.round(total/gt.goal*100)):0;
    const done=total>=gt.goal;
    html+='<div style="background:#fff;border-radius:var(--radius);padding:16px;box-shadow:var(--shadow);margin-bottom:12px;border:2px solid '+(done?'var(--gold)':'#eee')+'">'
      +'<div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:6px">'
        +'<strong>'+gt.name+'</strong>'
        +'<span style="font-size:.78rem;color:var(--purple);font-weight:700">'+gt.status+'</span>'
      +'</div>'
      +'<div style="background:#f0f0f0;border-radius:20px;height:10px;overflow:hidden;margin-bottom:6px">'
        +'<div style="background:'+(done?'var(--gold)':'var(--purple)')+';height:100%;width:'+pct+'%;border-radius:20px"></div>'
      +'</div>'
      +'<div style="display:flex;justify-content:space-between;font-size:.85rem;margin-bottom:8px">'
        +'<span style="color:var(--purple);font-weight:700">'+total.toLocaleString()+' 별 / '+gt.goal.toLocaleString()+' 별 ('+pct+'%)</span>'
        +'<span style="color:#888">'+myParts.length+'명 참여</span>'
      +'</div>'
      +'<div style="display:flex;gap:6px">'
        +(gt.status==='진행중'?'<button class="bsm bgr" style="padding:3px 9px;font-size:.77rem" onclick="doUpdateGtStatus(\''+gt.id+'\',\'완료\')">완료 처리</button>':'')
        +'<button class="bsm bre" style="padding:3px 9px;font-size:.77rem" onclick="doDeleteGt(\''+gt.id+'\')">삭제</button>'
      +'</div>'
    +'</div>';
  }
  el.innerHTML=html;
}
